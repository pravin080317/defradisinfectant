pageViews
| extend 
    pageTitle = tostring(customDimensions["pageTitle"]),
    pageName = iff(isnull(name) or name == "", tostring(url), name)
| project 
    timestamp,
    user_Id,
    session_Id,
    pageName,
    pageTitle,
    url
| summarize PageViewCount = count() by user_Id, session_Id, pageName, pageTitle, url
| where PageViewCount > 1
| order by user_Id, session_Id, PageViewCount desc
