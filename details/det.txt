function updatePageHeadingFromTable() {
  // count every row in the tbody of the ready-for-upload table
  const rowCount = document
    .querySelectorAll('#hideshow tbody.govuk-table__body tr')
    .length;

  const heading = document.getElementById('page-heading');
  heading.textContent =
    rowCount === 0
      ? 'Upload multiple reports'
      : `Upload multiple reports â€“ ${rowCount} document${rowCount > 1 ? 's' : ''} ready`;
}
