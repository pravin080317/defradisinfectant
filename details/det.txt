SELECT 
    a.Name AS AccountName,
    c.FullName AS ContactName,
    c.ContactId AS ContactId,
    p.Email AS PortalUserEmail,
    p.new_portalusersetupid AS PortalUserId
FROM 
    Contact c
INNER JOIN 
    new_portalusersetup p ON c.EmailAddress1 = p.Email
INNER JOIN 
    Account a ON a.AccountId = c.ParentCustomerId
WHERE 
    p.Email IN ('user1@example.com', 'user2@example.com') -- Add your imported users' emails here
