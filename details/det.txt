SELECT
    pr.[voap_billingauthorityaccount] AS CouncilCode,
    pfm.voap_uploadtypename AS UploadType,
    COUNT(*) AS FileCount
FROM
    voap_portalfilemanagement AS pfm
LEFT JOIN
    voap_portalrequest AS pr
    ON pfm.voap_portalrequestlookup = pr.voap_portalrequestid
WHERE
    pfm.statecode = 0
GROUP BY
    pr.[voap_billingauthorityaccount],
    pfm.voap_uploadtypename
ORDER BY
    CouncilCode, UploadType;
