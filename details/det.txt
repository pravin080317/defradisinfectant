function overridePaginationLinkColors() {
  const links = document.querySelectorAll('.govuk-pagination_link');

  links.forEach(link => {
    // Only override style if itâ€™s not the current page
    if (!link.closest('.govuk-pagination_item--current')) {
      // Force the color inline to avoid :visited override
      link.style.color = '#1d70b8'; // GOV.UK standard link color
      link.style.cursor = 'pointer'; // Restore hand pointer on hover
    }
  });
}
