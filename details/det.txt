customEvents
| where name == "VOAListScheduleDownload"
| extend 
    fileName = tostring(customDimensions.fileName),
    fileType = tostring(customDimensions.fileType),
    fileSize = toint(customDimensions.fileSize),
    userId = tostring(customDimensions.userId),
    userName = tostring(customDimensions.userName),
    council = tostring(customDimensions.council),
    eventTime = timestamp
| project 
    eventTime,
    fileName,
    fileType,
    fileSize,
    userId,
    userName,
    council
| order by eventTime desc
