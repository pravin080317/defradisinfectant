// Save initial active council info to sessionStorage
try {
  const existingCouncilId = sessionStorage.getItem("activeCouncilId");
  if (!existingCouncilId) {
    sessionStorage.setItem("activeCouncilId", "{{ user.voap_contact_ActiveAccount_account.accountid }}");
    sessionStorage.setItem("activeCouncilName", "{{ user.voap_contact_ActiveAccount_account.voa_baccodeacc }}");
  }
} catch (e) {
  console.error("Failed to set initial council info in sessionStorage", e);
}
