  // ðŸ” Add error ID to aria-describedby
  const fieldset = document.querySelector(`[aria-describedby*="${id}-hint"]`);
  if (fieldset && !fieldset.getAttribute("aria-describedby").includes(`${id}-error`)) {
    fieldset.setAttribute(
      "aria-describedby",
      fieldset.getAttribute("aria-describedby") + ` ${id}-error`
    );
  }

function clearAccessibleError(id) {
  const errorNode = document.getElementById(`${id}-error`);
  if (errorNode) errorNode.remove();

  // ðŸ” Remove from aria-describedby
  const fieldset = document.querySelector(`[aria-describedby*="${id}-hint"]`);
  if (fieldset) {
    const currentDesc = fieldset.getAttribute("aria-describedby");
    const updatedDesc = currentDesc
      .split(" ")
      .filter(d => d !== `${id}-error`)
      .join(" ");
    fieldset.setAttribute("aria-describedby", updatedDesc);
  }
}
