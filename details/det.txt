SELECT 
    fdh.voa_frontdoorheaderid,
    fdh.voa_portalfilemanagementid,
    pfm.voap_filename,
    pfm.statuscodename
FROM voa_frontdoorheader fdh
INNER JOIN voap_portalfilemanagement pfm
    ON fdh.voa_portalfilemanagementid = pfm.voap_portalfilemanagementid
WHERE fdh.voa_portalfilemanagementid IS NOT NULL
ORDER BY pfm.createdon DESC;
