function getLowercaseExtensionFilename(file) {
    const nameParts = file.name.split('.');
    
    // If no extension or hidden file (e.g., .gitignore), return as-is
    if (nameParts.length < 2) return file.name;

    const ext = nameParts.pop().toLowerCase(); // lowercase the extension
    const baseName = nameParts.join('.');      // rejoin rest of filename

    return `${baseName}.${ext}`;
}
