private string NormalizeCellValue(string value)
{
    if (double.TryParse(value, System.Globalization.NumberStyles.Float, System.Globalization.CultureInfo.InvariantCulture, out double numericValue))
    {
        // If the value looks like scientific notation (e.g., "2.00E+11"), convert it to a plain string (e.g., "200000000000")
        return numericValue.ToString("0", System.Globalization.CultureInfo.InvariantCulture);
    }
    return value;
}
