customEvents
| extend fileName = tostring(customDimensions.fileName)
| where isnotempty(fileName)
| extend 
    firstChar = substring(fileName, 0, 1),
    extension = extract(@"\.([a-zA-Z0-9]+)$", 1, fileName)
| where 
    firstChar == toUpper(firstChar) 
    or extension != tolower(extension)
| project timestamp, name, fileName, firstChar, extension
| order by timestamp desc
