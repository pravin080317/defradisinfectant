if ($Column -eq "UPRN" -and $Value) {
    # Ensure the value is treated as a string to preserve formatting
    $Value = $Value.Trim()

    # Check if UPRN is purely numeric (no decimals allowed)
    if ($Value -match "^\d+(\.\d+)?$") {
        if ($Value -match "\.") {
            $Errors += "$Column must be a whole number (no decimals allowed)"
        } else {
            # If it's in scientific notation, convert it to a whole number string
            try {
                $Value = [math]::Round([double]$Value, 0).ToString()  # Convert to string without decimals
            } catch {
                $Errors += "$Column contains invalid numeric format"
            }
        }
    } else {
        $Errors += "$Column must be numeric"
    }
}
