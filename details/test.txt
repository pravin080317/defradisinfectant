if ($Column -eq "UPRN" -and $Value) {
    # Remove any whitespace
    $Value = $Value.Trim()

    # Check if UPRN contains only digits (no decimals allowed)
    if ($Value -match "^\d+$") {
        # If it's in scientific notation, convert it
        try {
            $Value = [math]::Round([double]$Value, 0)  # Convert to whole number
        } catch {
            $Errors += "$Column contains invalid numeric format"
        }
    } else {
        $Errors += "$Column must be a whole number (no decimals allowed)"
    }
}
