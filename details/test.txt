if ($Column -eq "Date of Change" -and $Rules.ContainsKey("date_format") -and $Rules.date_format -and $Value) {
    try {
        # Convert Excel OADate to a proper date
        if ($Value -match "^\d+$") {
            $ParsedDate = [datetime]::FromOADate([double]$Value)
        } else {
            # Parse the date correctly, assuming it is in "dd-MMM-yy" format
            $ParsedDate = [datetime]::ParseExact($Value, "dd-MMM-yy", $null)
            
            # If the parsed year is before 2000, assume it's in 2000+ range
            if ($ParsedDate.Year -lt 2000) {
                $ParsedDate = $ParsedDate.AddYears(100)  # Fix PowerShell treating 24 as 1924 instead of 2024
            }
        }

        # Convert min_date to a proper datetime format
        if ($Rules.ContainsKey("min_date")) {
            $MinDate = [datetime]::ParseExact($Rules.min_date, "dd-MMM-yyyy", $null)  # Use full year format
            
            # Validate that the given date is not earlier than allowed
            if ($ParsedDate -lt $MinDate) {
                $Errors += "$Column must be after $($Rules.min_date)"
            }
        }
    } catch {
        $Errors += "$Column must be in format dd-MMM-yy"
    }
}
